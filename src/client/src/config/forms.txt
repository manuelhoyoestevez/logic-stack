var Forms = {
    auth: {
        className: clase para el contenedor del componente base del formulario,
        fields: [
            {
                type: tipo de campo del formulario (text, password, select, date, date-range, title, checkbox, textarea),
                id: id del campo,
                label: clave del archivo de traducciones para el label del campo,
                default: clave del archivo de traducciones para la opci칩n por defecto del selector (select),
                component: define el componente HTML de texto para cabeceras (title),
                placeholder: clave del archivo de traducciones para el texto predefinido del campo (text, password, textarea),
                className: clase para el input del campo,
                sx: nombre de la variable de estilos en el archivo styles.js de los master fields para a침adir estilos inline sin css,
                validation: {
                    validators: array de nombres de las funciones validadoras en el archivo validations.js,
                    message: clave del archivo de traducciones para el mensaje de error del campo
                },
                itemProps: {
                    propiedades para el componente Grid que envuelve el campo
                },
                inputProps: {
                    propiedades para el componente del campo
                },
                labelProps: {
                    propiedades para el label del campo
                },
                options: array de opciones del selector (select)
                [{
                    id: id de la opci칩n del selector,
                    value: valor de la opci칩n del selector
                }...]
            }
        ],
        actionsProps: {
            propiedades para el componente Grid que envuelve el bloque de las acciones
        }
    }
};
